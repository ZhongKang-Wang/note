- [ ] 栈的压入、弹出序列

```c++
bool isPopOrder(vector<int> pushV,vector<int> popV) {
        if (pushV.size() != popV.size()) {
            return false;
        }
        if (pushV.empty()) {
            return true;
        }
        int j = 0;
        int n = pushV.size();
        stack<int> st;
        for (int i = 0; i < n; ++i) {
            st.push(pushV[i]);
            while (!st.empty() && st.top() == popV[j]) {
                st.pop();
                ++j;
            }
        }
        return j == n;
}
```
